$csvcontent = Import-CSV -Path "C:\Usuarios.csv"
foreach ($user in $csvcontent)
{
New-ADUser-SamAccountName ($user.Usuario)
-AccountPassword (ConvertTo-SecureString “P4ssw0rd”

-givenname $user.Nombres
-surname $user.Apellidos

-Office $user.Direccion
-OfficePhone $user.Telefono

-State $user.State

-Department $user.Division
